006  (defn ensure-map
007    "Ensures `x`to be a map. If `x` is a value a map is constucted from
008    the last keyword and the value.
009  
010    Example:
011    ```clojure
012    (ensure-map 10 [:a :b :c])
013    ;; =>
014    ;; [{:c 10} [:a :b]]
015    (ensure-map {:d 10} [:a :b :c])
016    ;; =>
017    ;; [{:d 10} [:a :b :c]]
018    ```
019    "
020    [x v]
021    (if (map? x)
022      [x v]
023      [{(last v) x} (into [] (butlast v))]))
024  
025  (defn ensure-vec
026    "Ensures that `v` is a vector.
027  
028    Example:
029    ```clojure
030    (ensure-vec nil) ;!
031    ;; nil
032    (ensure-vec 1)
033    ;; [1]
034    (ensure-vec [1])
035    ;; [1]
036    ```"
037    [v]
038    (when v (if (vector? v) v [v])))

040  (defn vector-if
041    "Makes the value `v` behind the keyword `kw` a vector if `v` is not
042    nil."
043    [m kw]
044    (if (and (map? m) (keyword? kw))
045      (if-let [v (kw m)] (assoc m kw (ensure-vec v)) m)))

047  (defn replace-if
048    "Replaces `v`alue of `k`ey in struct if `v`is not `nil`.
049  
050    Example:
051    ```clojure
052    (replace-if {:Type \"a\"} :Type \"b\")
053    ;; {:Type \"b\"}
054    ```
055    "
056    [m k v]
057    (if (and (some? v) (keyword? k)) (assoc m k v) m))

059  (defn append-if
060    "Appends `v` to the value of `k`. If `k` does not exist in `m`,
061    `k [v]` is assoced.  If `k` does exist in `m`, `v` is conjed.
062  
063    Example:
064    ```clojure
065    (append-if {:Value [1 2 3]} :Value 4)
066    ;; {:Value [1 2 3 4]}
067    ```"
068    [m k v]
069    (if (and (some? v) (keyword? k))
070      (let [new-v (ensure-vec v)]
071        (if-let [old-v (k m)]
072          (assoc m k (into [] (concat old-v new-v)))
073          (assoc m k new-v)))
074      m))

076  (defn path->kw-vec
077    "Turns the path into a vector of keywords.
078  
079    Example:
080    ```clojure
081    (path->kw-vec \"a.b.c\")
082    ;; [:a :b :c]
083    ```"
084    [s]
085    {:pre [(string? s)]}
086    (mapv keyword (string/split s (re-pattern "\\."))))
